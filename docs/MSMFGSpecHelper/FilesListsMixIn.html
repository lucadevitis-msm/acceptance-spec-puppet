def file_list(pattern)
  FileList[pattern].tap do |list|
    list.exclude 'vendor/**/*',       # bundler
                 'pkg/**/*',          # gem build process
                 'spec/fixtures/**/*' # puppetlabs fixtures
    list.reject! { |f| File.directory? f }
  end
end
